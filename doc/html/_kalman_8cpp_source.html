159  void stateConversionFunction(const cv::Mat& x_k, const cv::Mat& u_k, const cv::Mat& v_k, cv::Mat& x_kplus1)
160  {
161  track_t x0 = x_k.at<track_t>(0, 0);
162  track_t y0 = x_k.at<track_t>(1, 0);
163  track_t vx0 = x_k.at<track_t>(2, 0);
164  track_t vy0 = x_k.at<track_t>(3, 0);
165  track_t ax0 = x_k.at<track_t>(4, 0);
166  track_t ay0 = x_k.at<track_t>(5, 0);
167 
168  x_kplus1.at<track_t>(0, 0) = x0 + vx0 * m_deltaTime + ax0 * sqr(m_deltaTime) / 2;
169  x_kplus1.at<track_t>(1, 0) = y0 + vy0 * m_deltaTime + ay0 * sqr(m_deltaTime) / 2;
170  x_kplus1.at<track_t>(2, 0) = vx0 + ax0 * m_deltaTime;
171  x_kplus1.at<track_t>(3, 0) = vy0 + ay0 * m_deltaTime;
172  x_kplus1.at<track_t>(4, 0) = ax0;
173  x_kplus1.at<track_t>(5, 0) = ay0;
174 
175  if (m_rectModel)
176  {
177  x_kplus1.at<track_t>(6, 0) = x_k.at<track_t>(6, 0);
178  x_kplus1.at<track_t>(7, 0) = x_k.at<track_t>(7, 0);
179  }
180 
181  if (v_k.size() == u_k.size())
182  {
183  x_kplus1 += v_k + u_k;
184  }
185  else
186  {
187  x_kplus1 += v_k;
188  }
189  }

445 //---------------------------------------------------------------------------
446 Point_t TKalmanFilter::Update(Point_t pt, bool dataCorrect)
447 {
448  if (!m_initialized)
449  {
450  if (m_initialPoints.size() < MIN_INIT_VALS)
451  {
452  if (dataCorrect)
453  {
454  m_initialPoints.push_back(pt);
455  }
456  }
457  if (m_initialPoints.size() == MIN_INIT_VALS)
458  {
459  track_t kx = 0;
460  track_t bx = 0;
461  track_t ky = 0;
462  track_t by = 0;
463  get_lin_regress_params(m_initialPoints, 0, MIN_INIT_VALS, kx, bx, ky, by);
464  Point_t xy0(kx * (MIN_INIT_VALS - 1) + bx, ky * (MIN_INIT_VALS - 1) + by);
465  Point_t xyv0(kx, ky);
466 
467  switch (m_type)
468  {
469  case tracking::KalmanLinear:
470  CreateLinear(xy0, xyv0);
471  break;
472 
473  case tracking::KalmanUnscented:
474 #ifdef USE_OCV_UKF
475  CreateUnscented(xy0, xyv0);
476 #else
477  CreateLinear(xy0, xyv0);
478  std::cerr << "UnscentedKalmanFilter was disabled in CMAKE! Set KalmanLinear in constructor." << std::endl;
479 #endif
480  break;
481 
482  case tracking::KalmanAugmentedUnscented:
483 #ifdef USE_OCV_UKF
484  CreateAugmentedUnscented(xy0, xyv0);
485 #else
486  CreateLinear(xy0, xyv0);
487  std::cerr << "AugmentedUnscentedKalmanFilter was disabled in CMAKE! Set KalmanLinear in constructor." << std::endl;
488 #endif
489  break;
490  }
491  }
492  }
493 
494  if (m_initialized)
495  {
496  cv::Mat measurement(2, 1, Mat_t(1));
497  if (!dataCorrect)
498  {
499  measurement.at<track_t>(0) = m_lastPointResult.x; //update using prediction
500  measurement.at<track_t>(1) = m_lastPointResult.y;
501  }
502  else
503  {
504  measurement.at<track_t>(0) = pt.x; //update using measurements
505  measurement.at<track_t>(1) = pt.y;
506  }
507  // Correction
508  cv::Mat estimated;
509  switch (m_type)
510  {
511  case tracking::KalmanLinear:
512  estimated = m_linearKalman->correct(measurement);
513  break;
514 
515  case tracking::KalmanUnscented:
516  case tracking::KalmanAugmentedUnscented:
517 #ifdef USE_OCV_UKF
518  estimated = m_uncsentedKalman->correct(measurement);
519 #else
520  estimated = m_linearKalman->correct(measurement);
521  std::cerr << "UnscentedKalmanFilter was disabled in CMAKE! Set KalmanLinear in constructor." << std::endl;
522 #endif
523  break;
524  }
525 
526  m_lastPointResult.x = estimated.at<track_t>(0); //update using measurements
527  m_lastPointResult.y = estimated.at<track_t>(1);
528  }
529  else
530  {
531  if (dataCorrect)
532  {
533  m_lastPointResult = pt;
534  }
535  }
536  return m_lastPointResult;
537 }
