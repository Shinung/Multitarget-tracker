91

93 
94  if (level_number) {
95  level_number->init (G);
96  }
97

99  preds->init (G, node());
100  }
101

103

106  }
107

119

122  node curr;
123

125  forall_nodes (curr, G) {

133  }
134

137  }
138 
139  end_handler (G);
140

151 void bfs::bfs_sub (graph& G, const node& st, edge_map<int>* used)
152 {
153  qu.push_back (st);
154  bfs_number[st] = act_bfs_num;
155  ++act_bfs_num;
156 
157  if (level_number) {
158  (*level_number)[st] = 0;
159  }
160 
161  while (!qu.empty()) {
162  node tmp = qu.front();
163  qu.pop_front();
164  ++reached_nodes;
165 
166  if (tmp == st) {
167  roots.push_back (bfs_order.insert (bfs_order.end(), tmp));
168  } else {
169  bfs_order.push_back (tmp);
170  }
171 
172  popped_node_handler (G, tmp);
173 
174  node::adj_edges_iterator it = tmp.adj_edges_begin();
175  node::adj_edges_iterator end = tmp.adj_edges_end();
176 
177  for (; it != end; ++it) {
178  edge curr = *it;
179  node opp = tmp.opposite (curr);
180 
181  if (bfs_number[opp] == 0) {
182  bfs_number[opp] = act_bfs_num;
183  ++act_bfs_num;
184  tree.push_back (curr);
185 
186  if (non_tree) {
187  (*used)[curr] = 1;
188  }
189 
190  if (level_number) {
191  (*level_number)[opp] = (*level_number)[tmp] + 1;
192  }
193 
194  if (preds) {
195  (*preds)[opp] = tmp;
196  }
197 
198  qu.push_back (opp);
199 
200  unused_node_handler (G, opp, tmp);
201 
202  } else {
203  if (non_tree && !(*used)[curr]) {
204  (*used)[curr] = 1;
205  non_tree->push_back(curr);
206  }
207 
208  used_node_handler (G, opp, tmp);
209  }
210  }
211 
212  finished_node_handler (G, tmp);
213  }
214 }
