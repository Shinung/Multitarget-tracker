58 void biconnectivity::init_handler(graph& G)
59 {
60  if (add_edges) {
61  dfs D;
62  D.scan_whole_graph(true);
63  D.check(G);
64  D.run(G);
65 
66  roots_iterator it, end;
67  it = D.roots_begin();
68  end = D.roots_end();
69  start = *(*it);
70  ++it;
71 
72  for (; it != end; ++it) {
73  additional.push_back(G.new_edge(start, *(*it)));
74  }
75 
76  first_child.init(G, node());
77  }
78 
79  low_num.init(G);
80  in_component.init(G);
81  cut_count.init(G, 0);
82 
83  //
84  // Detect self loops and hide them.
85  //
86 
87  assert(self_loops.empty());
88  graph::edge_iterator eit = G.edges_begin(),
89  eend = G.edges_end();
90 
91  while (eit != eend) {
92  edge e = *eit;
93  eit++;
94  if (e.target() == e.source()) {
95  self_loops.push_back(e);
96  G.hide_edge(e);
97  }
98  }
99 }

114 void biconnectivity::new_start_handler(graph& /*G*/, node& st)
115 {
116  cut_count[st] = -1;
117 
118  //
119  // If this node has no adjacent edges, we
120  // must write down the component right here. This is because
121  // then the method after_recursive_call_handle is never
122  // executed.

124  // 28/2/2002 MR
125  //
126 
127  if (st.degree() == 0) {
128  ++num_of_components;

149 void biconnectivity::after_recursive_call_handler(graph& G, edge& e, node& n)
150 {
151  node curr = n.opposite(e);
152 
153  if (low_num[n] < low_num[curr]) {
154  low_num[curr] = low_num[n];
155  }
156 
157  if (low_num[n] >= dfs_num(curr)) {
158  //
159  // Component found
160  //
161 
162  if (store_comp) {
163  component_iterator li = components.insert(
164  components.end(),
165  std::pair<nodes_t, edges_t>(nodes_t(), edges_t()));
166 
167  nodes_t& component = li->first;
168  edges_t& co_edges = li->second;
169 
170  //
171  // Nodes of biconnected component
172  //
173 
174  node tmp = node_stack.top();
175 
176  while (dfs_num(tmp) >= dfs_num(n)) {
177  node_stack.pop();
178  component.push_back(tmp);
179  in_component[tmp] = li;
180  if (node_stack.empty()) break;
181  else tmp = node_stack.top();
182  }
183 
184  component.push_back(curr);
185 
186  //
187  // edges of biconnected component
188  //
189 
190  edge ed = edge_stack.top();
191 
192  while ((dfs_num(ed.source()) >= dfs_num(n) &&
193  dfs_num(ed.target()) >= dfs_num(n)) ||
194  (dfs_num(ed.source()) == dfs_num(curr) &&
195  dfs_num(ed.target()) >= dfs_num(n)) ||
196  (dfs_num(ed.source()) >= dfs_num(n) &&
197  dfs_num(ed.target()) == dfs_num(curr))) {
198  edge_stack.pop();
199  co_edges.push_back(ed);
200  if (edge_stack.empty()) break;
201  else ed = edge_stack.top();
202  }
203  }
204 
205 
206  ++num_of_components;
207 
208  //
209  // curr is cut point; increase counter
210  //
211 
212  ++cut_count[curr];
213 
214  if (add_edges) {
215  node father = (*preds)[curr];
216  node first = first_child[curr];
217 
218  if (father != node() && n == first) {
219  additional.push_back(G.new_edge(father, first));
220  }
221 
222  if (n != first) {
223  additional.push_back(G.new_edge(n, first));
224  }
225  }
226 
227  }
228 }

257 void biconnectivity::end_handler(graph& G)
258 {
259  edges_t::iterator it = self_loops.begin();
260  edges_t::iterator end = self_loops.end();
261 
262  while (it != end)
263  {
264  G.restore_edge(*it);
265  if (store_comp)
266  {
267  component_iterator cit = in_component[it->target()];
268  cit->second.push_back(*it);
269  }
270 
271  it = self_loops.erase(it);
272  }
273 }
