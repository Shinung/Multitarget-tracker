319 void IntImage<T>::Resize(IntImage<T>& result,const int height,const int width) const
320 {
321  assert(height>0 && width>0);
322  result.SetSize(height,width);
323  REAL ixratio = nrow*1.0/height, iyratio = ncol*1.0/width;
324 
325  REAL* p_y = new REAL[result.ncol];
326  assert(p_y!=NULL);
327  int* p_y0 = new int[result.ncol];
328  assert(p_y0!=NULL);
329  for(int i=0; i<width; i++)
330  {
331  p_y[i] = i*iyratio;
332  p_y0[i] = (int)p_y[i];
333  if(p_y0[i]==ncol-1) p_y0[i]--;
334  p_y[i] -= p_y0[i];
335  }
336 
337  for(int i=0; i<height; i++)
338  {
339  int x0;
340  REAL x;
341  x = i*ixratio;
342  x0 = (int)x;
343  if(x0==nrow-1) x0--;
344  x -= x0;
345  T* rp = result.p[i];
346  const T* px0 = p[x0];
347  const T* px1 = p[x0+1];
348  for(int j=0; j<width; j++)
349  {
350  int y0=p_y0[j];
351  REAL y=p_y[j],fx0,fx1;
352 
353  fx0 = REAL(px0[y0] + y*(px0[y0+1]-px0[y0]));
354  fx1 = REAL(px1[y0] + y*(px1[y0+1]-px1[y0]));
355 
356  rp[j] = T(fx0 + x*(fx1-fx0));
357  }
358  }
359

396 template<class T>
397 void IntImage<T>::Sobel(IntImage<REAL>& result,const bool useSqrt,const bool normalize)
398 {
399  // compute the Sobel gradient. For now, we just use the very inefficient way. Optimization can be done later
400 // if useSqrt = true, we compute the real Sobel gradient; otherwise, the square of it
401 // if normalize = true, the numbers are normalized to be in 0..255
402  result.Create(nrow,ncol);
403  for(int i=0; i<nrow; i++) result.p[i][0] = result.p[i][ncol-1] = 0;
404  std::fill(result.p[0],result.p[0]+ncol,0.0);
405  std::fill(result.p[nrow-1],result.p[nrow-1]+ncol,0.0);
406  for(int i=1; i<nrow-1; i++)
407  {
408  T* p1 = p[i-1];
409  T* p2 = p[i];
410  T* p3 = p[i+1];
411  REAL* pr = result.p[i];
412  for(int j=1; j<ncol-1; j++)
413  {
414  REAL gx = p1[j-1] - p1[j+1]
415  + 2*(p2[j-1] - p2[j+1])
416  + p3[j-1] - p3[j+1];
417  REAL gy = p1[j-1] - p3[j-1]
418  + 2*(p1[j] - p3[j])
419  + p1[j+1] - p3[j+1];
420  pr[j] = gx*gx+gy*gy;
421  }
422  }
423  if(useSqrt || normalize ) // if we want to normalize the result image, we'd better use the true Sobel gradient
424  for(int i=1; i<nrow-1; i++)
425  for(int j=1; j<ncol-1; j++)
426  result.p[i][j] = sqrt(result.p[i][j]);
427 
428  if(normalize)
429  {
430  REAL minv = 1e20, maxv = -minv;
431  for(int i=1; i<nrow-1; i++)
432  {
433  for(int j=1; j<ncol-1; j++)
434  {
435  if(result.p[i][j]<minv)
436  minv = result.p[i][j];
437  else if(result.p[i][j]>maxv)
438  maxv = result.p[i][j];
439  }
440  }
441  for(int i=0; i<nrow; i++) result.p[i][0] = result.p[i][ncol-1] = minv;
442  for(int i=0; i<ncol; i++) result.p[0][i] = result.p[nrow-1][i] = minv;
443  REAL s = 255.0/(maxv-minv);
444  for(int i=0; i<nrow*ncol; i++) result.buf[i] = (result.buf[i]-minv)*s;
445  }
446 }
