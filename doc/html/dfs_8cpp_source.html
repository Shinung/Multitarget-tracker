66 int dfs::run (graph& G)
67 {
68  //
69  // initialization
70  //
71 
72  node curr;
73  node dummy;
74 
75  dfs_number.init (G, 0);
76 
77  if (comp_number) {
78  comp_number->init (G);
79  }
80 
81  if (preds) {
82  preds->init (G, node());
83  }
84 
85  if (back_edges) {
86  used = new edge_map<int> (G, 0);
87  }
88 
89  init_handler (G);
90 
91  //
92  // Set start-node
93  //
94 
95  if (G.number_of_nodes() == 0) {
96  return GTL_OK;
97  }
98 
99  if (start == node()) {
100  start = G.choose_node();
101  }
102 
103  new_start_handler (G, start);
104 
105  dfs_sub (G, start, dummy);
106 
107  if (whole_graph && reached_nodes < G.number_of_nodes()) {
108 
109  //
110  // Continue DFS with next unused node.
111  //
112 
113  forall_nodes (curr, G) {
114  if (dfs_number[curr] == 0) {
115  new_start_handler (G, curr);
116  dfs_sub (G, curr, dummy);
117  }
118  }
119  }
120 
121  if (back_edges) {
122  delete used;
123  used = 0;
124  }
125 
126  end_handler(G);
127 
128  return GTL_OK;
129 }

137 void dfs::dfs_sub (graph& G, node& curr, node& father)
138 {
139  node opp;
140  edge adj;
141 
142  if (father == node()) {
143  roots.push_back (dfs_order.insert (dfs_order.end(), curr));
144  } else {
145  dfs_order.push_back (curr);
146  }
147 
148  dfs_number[curr] = act_dfs_num;
149  reached_nodes++;
150 
151  if (preds) {
152  (*preds)[curr] = father;
153  }
154 
155  entry_handler (G, curr, father);
156 
157  ++act_dfs_num;
158  node::adj_edges_iterator it = curr.adj_edges_begin();
159  node::adj_edges_iterator end = curr.adj_edges_end();
160 
161  while (it != end) {
162  adj = *it;
163  opp = curr.opposite(adj);
164 
165  if (dfs_number[opp] == 0) {
166  tree.push_back (adj);
167 
168  if (back_edges) {
169  (*used)[adj] = 1;
170  }
171 
172  before_recursive_call_handler (G, adj, opp);
173  dfs_sub (G, opp, curr);
174  after_recursive_call_handler (G, adj, opp);
175 
176  } else {
177  if (back_edges && !(*used)[adj]) {
178  (*used)[adj] = 1;
179  back_edges->push_back (adj);
180  }
181 
182  old_adj_node_handler (G, adj, opp);
183  }
184 
185  ++it;
186  }
187 
188  leave_handler (G, curr, father);
189 
190  if (comp_number) {
191  (*comp_number)[curr] = act_comp_num;
192  ++act_comp_num;
193  }
194 }
