107 int dijkstra::check(graph& G)
108 {
109  if ((s == node()) || (!weights_set))
110  {
111  return GTL_ERROR;
112  }
113 
114  bool source_found = false;
115  graph::node_iterator node_it;
116  graph::node_iterator nodes_end = G.nodes_end();
117  for (node_it = G.nodes_begin(); node_it != nodes_end; ++node_it)
118  {
119  if (*node_it == s)
120  {
121  source_found = true;
122  break;
123  }
124  }
125  if (!source_found)
126  {
127  return(GTL_ERROR);
128  }
129 
130  graph::edge_iterator edge_it;
131  graph::edge_iterator edges_end = G.edges_end();
132  for(edge_it = G.edges_begin(); edge_it != edges_end; ++edge_it)
133  {
134  if (weight[*edge_it] < 0.0)
135  {
136  return false;
137  }
138  }
139 
140  return GTL_OK;

144 int dijkstra::run(graph& G)
145 {
146  init(G);
147 
148  less_dist prd(&dist, &mark);
149  bin_heap<node, less_dist> node_heap(prd, G.number_of_nodes());
150  mark[s] = grey;
151  dist[s] = 0.0;
152  node_heap.push(s);
153  while (!node_heap.is_empty())
154  {
155 
156  // debug:
157  // node_heap.print_data_container();
158 
159  node cur_node = node_heap.top();
160  node_heap.pop();
161 
162  // debug:
163  // node_heap.print_data_container();
164 
165  mark[cur_node] = white;
166  if (cur_node == t)
167  {
168  // if @a t is set through #target we are ready
169  return GTL_OK;
170  }
171 
172  node::adj_edges_iterator adj_edge_it;
173  node::adj_edges_iterator adj_edges_end = cur_node.adj_edges_end();
174  for (adj_edge_it = cur_node.adj_edges_begin();
175  adj_edge_it != adj_edges_end;
176  ++adj_edge_it)
177  {
178  node op_node = adj_edge_it->opposite(cur_node);
179  if (mark[op_node] == black)
180  {
181  mark[op_node] = grey;
182  dist[op_node] = dist[cur_node] + weight[*adj_edge_it];
183  node_heap.push(op_node);
184 
185  // debug:
186  // node_heap.print_data_container();
187 
188  if (preds_set)
189  {
190  pred[op_node] = *adj_edge_it;
191  }
192  }
193  else if (mark[op_node] == grey)
194  {
195  if (dist[op_node] > dist[cur_node] + weight[*adj_edge_it])
196  {
197  dist[op_node] = dist[cur_node] + weight[*adj_edge_it];
198  node_heap.changeKey(op_node);
199 
200  // debug:
201  // node_heap.print_data_container();
202 
203  if (preds_set)
204  {
205  pred[op_node] = *adj_edge_it;
206  }
207  }
208  }
209  else // (mark[op_node] == white)
210  {
211  // nothing to do: shortest distance to op_node is already
212  // computed
213  }
214  }
215  }
216 
217  return GTL_OK;
218 }
