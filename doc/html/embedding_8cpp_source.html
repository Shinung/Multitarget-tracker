154 bool
155 planar_embedding::check ()
156 {
157  node n;
158  forall_nodes (n ,*G) {
159  iterator it, end;
160 
161  for (it = adj[n].begin(), end = adj[n].end(); it != end; ++it) {
162  edge curr = *it;
163  node other = n.opposite (curr);
164 
165  edge prev = cyclic_prev (n, curr);
166  edge next = cyclic_next (n, prev);
167  assert (next == curr);
168 
169  while (other != n) {
170  curr = cyclic_next (other, curr);
171  other = other.opposite (curr);
172  }
173  if (curr != prev) {
174  return false;
175  }
176 
177  }
178  }
179 
180  return true;
181 }
