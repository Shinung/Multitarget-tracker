136 /* fh_delete_min() - deletes the minimum node from the heap pointed to by h and
137  * returns its vertex number.
138  */
139 int fh_delete_min(fheap_t *h)
140 {
141  fheap_node_t *min_node, *child, *next;
142  long k, k2;
143  int r, v, vertex_no;
144 
145 #if FHEAP_DUMP
146 printf("delete_min, ");
147 #endif
148 
149  /* First we determine the maximum rank in the heap. */
150  v = h->value;
151  r = -1;
152  while(v) {
153  v = v >> 1;
154  r++;
155  };
156 
157  /* Now determine which root node is the minimum. */
158  min_node = h->trees[r];
159  k = min_node->key;
160  while(r > 0) {
161  r--;
162  next = h->trees[r];
163  if(next) {
164  if((k2 = next->key) < k) {
165  k = k2;
166  min_node = next;
167  }
168  h->key_comps++;
169  }
170  }
171 
172  /* We remove the minimum node from the heap but keep a pointer to it. */
173  r = min_node->rank;
174  h->trees[r] = NULL;
175  h->value -= (1 << r);
176 
177  child = min_node->child;
178  if(child) fh_meld(h, child);
179 
180  /* Record the vertex no of the old minimum node before deleting it. */
181  vertex_no = min_node->vertex_no;
182  h->nodes[vertex_no] = NULL;
183  free(min_node);
184  h->n--;
185 
186 #if FHEAP_DUMP
187 printf("delete_min-exited, ");
188 #endif
189 
190  return vertex_no;
191 }

194 /* fh_decrease_key() - decreases the key used for vertex, vertex_no, to
195  * new_value. No check is made to ensure that new_value is in-fact less than
196  * the current value so it is up to the user of this function to ensure that
197  * it is.
198  */
199 void fh_decrease_key(fheap_t *h, int vertex_no, long new_value)
200 {
201  fheap_node_t *cut_node, *parent, *new_roots, *r, *l;
202  int prev_rank;
203 
204 #if FHEAP_DUMP
205 printf("decrease_key on vn = %d, ", vertex_no);
206 #endif
207 
208  /* Obtain a pointer to the decreased node and its parent then decrease the
209  * nodes key.
210  */
211  cut_node = h->nodes[vertex_no];
212  parent = cut_node->parent;
213  cut_node->key = new_value;
214 
215  /* No reinsertion occurs if the node changed was a root. */
216  if(!parent) {
217 #if FHEAP_DUMP
218 printf("decrease_key-exited, ");
219 #endif
220  return;
221  }
222 
223  /* Update the left and right pointers of cut_node and its two neighbouring
224  * nodes.
225  */
226  l = cut_node->left;
227  r = cut_node->right;
228  l->right = r;
229  r->left = l;
230  cut_node->left = cut_node->right = cut_node;
231 
232  /* Initially the list of new roots contains only one node. */
233  new_roots = cut_node;
234 
235  /* While there is a parent node that is marked a cascading cut occurs. */
236  while(parent && parent->marked) {
237 
238  /* Decrease the rank of cut_node's parent an update its child pointer.
239  */
240  parent->rank--;
241  if(parent->rank) {
242  if(parent->child == cut_node) parent->child = r;
243  }
244  else {
245  parent->child = NULL;
246  }
247 
248  /* Update the cut_node and parent pointers to the parent. */
249  cut_node = parent;
250  parent = cut_node->parent;
251 
252  /* Update the left and right pointers of cut_nodes two neighbouring
253  * nodes.
254  */
255  l = cut_node->left;
256  r = cut_node->right;
257  l->right = r;
258  r->left = l;
259 
260  /* Add cut_node to the list of nodes to be reinserted as new roots. */
261  l = new_roots->left;
262  new_roots->left = l->right = cut_node;
263  cut_node->left = l;
264  cut_node->right = new_roots;
265  new_roots = cut_node;
266  }
267 
268  /* If the root node is being relocated then update the trees[] array.
269  * Otherwise mark the parent of the last node cut.
270  */
271  if(!parent) {
272  prev_rank = cut_node->rank + 1;
273  h->trees[prev_rank] = NULL;
274  h->value -= (1 << prev_rank);
275  }
276  else {
277  /* Decrease the rank of cut_node's parent an update its child pointer.
278  */
279  parent->rank--;
280  if(parent->rank) {
281  if(parent->child == cut_node) parent->child = r;
282  }
283  else {
284  parent->child = NULL;
285  }
286 
287  parent->marked = 1;
288  }
289 
290  /* Meld the new roots into the heap. */
291  fh_meld(h, new_roots);
292 
293 #if FHEAP_DUMP
294 printf("decrease_key-exited, ");
295 #endif
296 }

302 /* fh_meld() - melds the linked list of trees pointed to by *tree_list into
303  * the heap pointed to by h.
304  */
305 void fh_meld(fheap_t *h, fheap_node_t *tree_list)
306 {
307  fheap_node_t *first, *next, *node_ptr, *new_root, *temp, *temp2, *lc, *rc;
308  int r;
309 
310 #if FHEAP_DUMP
311 printf("meld: ");
312 #endif
313 
314  /* We meld each tree in the circularly linked list back into the root level
315  * of the heap. Each node in the linked list is the root node of a tree.
316  * The circularly linked list uses the sibling pointers of nodes. This
317  * makes melding of the child nodes from a delete_min operation simple.
318  */
319  node_ptr = first = tree_list;
320 
321  do {
322 
323 #if FHEAP_DUMP
324 printf("%d, ", node_ptr->vertex_no);
325 #endif
326 
327  /* Keep a pointer to the next node and remove sibling and parent links
328  * from the current node. node_ptr points to the current node.
329  */
330  next = node_ptr->right;
331  node_ptr->right = node_ptr->left = node_ptr;
332  node_ptr->parent = NULL;
333 
334  /* We merge the current node, node_ptr, by inserting it into the
335  * root level of the heap.
336  */
337  new_root = node_ptr;
338  r = node_ptr->rank;
339 
340  /* This loop inserts the new root into the heap, possibly restructuring
341  * the heap to ensure that only one tree for each degree exists.
342  */
343  do {
344 
345  /* Check if there is already a tree of degree r in the heap.
346  * If there is then we need to link it with new_root so it will be
347  * reinserted into a new place in the heap.
348  */
349  if((temp = h->trees[r])) {
350 
351  /* temp will be linked to new_root and relocated so we no
352  * longer will have a tree of degree r.
353  */
354  h->trees[r] = NULL;
355  h->value -= (1 << r);
356 
357  /* Swap temp and new_root if necessary so that new_root always
358  * points to the root node which has the smaller key of the
359  * two.
360  */
361  if(temp->key < new_root->key) {
362  temp2 = new_root;
363  new_root = temp;
364  temp = temp2;
365  }
366  h->key_comps++;
367 
368  /* Link temp with new_root, making sure that sibling pointers
369  * get updated if rank is greater than 0. Also, increase r for
370  * the next pass through the loop since the rank of new has
371  * increased.
372  */
373  if(r++ > 0) {
374  rc = new_root->child;
375  lc = rc->left;
376  temp->left = lc;
377  temp->right = rc;
378  lc->right = rc->left = temp;
379  }
380  new_root->child = temp;
381  new_root->rank = r;
382  temp->parent = new_root;
383  temp->marked = 0;
384  }
385  /* Otherwise if there is not a tree of degree r in the heap we
386  * allow new_root, which possibly carries moved trees in the heap,
387  * to be a tree of degree r in the heap.
388  */
389  else {
390 
391  h->trees[r] = new_root;
392  h->value += (1 << r);;
393 
394  /* NOTE: Because new_root is now a root we ensure it is
395  * marked.
396  */
397  new_root->marked = 1;
398  }
399 
400  /* Note that temp will be NULL if and only if there was not a tree
401  * of degree r.
402  */
403  } while(temp);
404 
405  node_ptr = next;
406 
407  } while(node_ptr != first);
408 
409 #if FHEAP_DUMP
410 printf("meld-exited, ");
411 #endif
412 }
