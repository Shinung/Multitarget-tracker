48 /* The structure type for Fibonacci heap nodes.
49  *
50  * Nodes have the following pointers:
51  * parent - a pointer to the nodes parent node (if any).
52  * child - a pointer to a child node (typically the highest rank child).
53  * left, right - sibling pointers which provide a circular doubly linked list
54  * containing all the parents nodes children.
55  *
56  * The remaining structure fields are:
57  * rank - the nodes rank, that is, the number of children it has.
58  * `key' - the nodes key.
59  * vertex_no - the number of the graph vertex that the node corresponds to.
60  * Vertex numbering in the graph should be:
61  * 1, 2, 3, ... max_vertex.
62  */
63 typedef struct fheap_node {
64  struct fheap_node *parent;
65  struct fheap_node *left, *right;
66  struct fheap_node *child;
67  int rank;
68  int marked;
69  long key;
70  int vertex_no;
71 } fheap_node_t;
72 
73 /* The structure type for a Fibonacci heap.
74  *
75  * trees - An array of pointers to trees at root level in the heap. Entry i
76  * in the array points to the root node of a tree that has nodes of
77  * dimension i on the main trunk.
78  * nodes - An array of pointers to nodes in the heap. Nodes are indexed
79  * according to their vertex number. This array can then be used to
80  * look up the node for corresponding to a vertex number, and is
81  * useful when freeing space taken up by the heap.
82  * max_nodes - The maximum number of nodes allowed in the heap.
83  * max_trees - The maximum number of trees allowed in the heap (calculated from
84  * max_nodes).
85  * n - The current number of nodes in the heap.
86  * value - The binary value represented by trees in the heap.
87  * By maintaining this it is easy to keep track of the maximum rank
88  * tree in the heap.
89  * key_comps - can be used for experimental purposes when counting the number
90  * of key comparisons.
91  */
92 typedef struct fheap {
93  fheap_node_t **trees;
94  fheap_node_t **nodes;
95  int max_nodes, max_trees, n, value;
96  long key_comps;
97 } fheap_t;
