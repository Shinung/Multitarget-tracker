18 void Tokenise (std::string s, std::string delimiters, std::vector<std::string> &tokens)
19 {
20  tokens.erase (tokens.begin(), tokens.end());
21  int start, stop;
22  int n = s.length();
23  start = s.find_first_not_of (delimiters);
24  while ((start >= 0) && (start < n))
25  {
26  stop = s.find_first_of (delimiters, start);
27  if ((stop < 0) || (stop > n)) stop = n;
28  tokens.push_back (s.substr(start, stop - start));
29  start = stop + delimiters.length();
30  }
31 }
