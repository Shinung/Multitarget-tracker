75 void graph::copy (const graph& G,
76  nodes_t::const_iterator it,
77  nodes_t::const_iterator end)
78 {
79  node_map<node> copy (G, node());
80  nodes_t::const_iterator n_it;
81  nodes_t::const_iterator n_end;
82 
83  for(n_it = it, n_end = end; n_it != n_end; ++n_it)
84  {
85  copy[*n_it] = new_node();
86  }
87 
88  for(n_it = it, n_end = end; n_it != n_end; ++n_it)
89  {
90  node::out_edges_iterator e_it, e_end;
91 
92  for (e_it = n_it->out_edges_begin(), e_end = n_it->out_edges_end();
93  e_it != e_end; ++e_it) {
94 
95  if (copy[e_it->target()] != node()) {
96  new_edge(copy[e_it->source()], copy[e_it->target()]);
97  }
98  }
99  }
100 }

254 void graph::del_node(node n)
255 {
256  assert (n.data);
257  assert (n.data->owner == this);
258 
259  // delete edges
260 
261  while(n.in_edges_begin() != n.in_edges_end())
262  {
263  del_edge (*n.in_edges_begin());
264  }
265 
266  while(n.out_edges_begin() != n.out_edges_end())
267  {
268  del_edge (*n.out_edges_begin());
269  }
270 
271  //
272  // delete hidden edges adjacent to n.
273  //
274  // [ TODO ] This is only a quick fix and should be thought
275  // over some time or the other.
276  //
277 
278  edges_t::iterator it = hidden_edges.begin();
279  edges_t::iterator end = hidden_edges.end();
280 
281  while (it != end)
282  {
283  if (it->source() == n || it->target() == n)
284  {
285  delete it->data;
286  it = hidden_edges.erase (it);
287  }
288  else
289  {
290  ++it;
291  }
292  }
293 
294  // delete node
295 
296  pre_del_node_handler(n);
297 
298  nodes.erase(n.data->pos);
299  --nodes_count;
300  free_node_ids.push_back(n.data->id);
301  ++free_node_ids_count;
302  delete n.data;
303 
304  post_del_node_handler();
305 }

385 bool graph::is_bidirected(edge_map<edge>& rev) const {
386  edge e1;
387  node target, source;
388  bool bidirected = true;
389  node::out_edges_iterator it;
390  node::out_edges_iterator end;
391 
392  forall_edges (e1, *this) {
393  target = e1.target ();
394  source = e1.source ();
395  end = target.out_edges_end ();
396  it = target.out_edges_begin ();
397 
398  //
399  // Search all out-edges of target if they are connected to the actual
400  // edges source.
401  //
402 
403  while (it != end) {
404  if (it->target () == source) {
405  break;
406  }
407  ++it;
408  }
409 
410  if (it == end) {
411  bidirected = false;
412  rev[e1] = edge ();
413  } else {
414  rev[e1] = *it;
415  }
416  }
417 
418  return bidirected;
419 }

512 void graph::hide_edge (edge e)
513 {
514  assert (e.data->owner == this);
515  assert (e.data->owner == this);
516 
517  pre_hide_edge_handler (e);
518 
519  if (!e.is_hidden()) {
520 
521  //
522  // remove e from all sources and targets adjacency lists
523  //
524  e.remove_from(0);
525  e.remove_from(1);
526 
527  //
528  // clear the list of positions
529  //
530  e.data->adj_pos[0].erase
531  (e.data->adj_pos[0].begin(), e.data->adj_pos[0].end());
532  e.data->adj_pos[1].erase
533  (e.data->adj_pos[1].begin(), e.data->adj_pos[1].end());
534 
535  //
536  // remove e from the list of all edges
537  //
538  edges.erase (e.data->pos);
539 
540  //
541  // insert e in hidden edges list
542  //
543  e.data->pos = hidden_edges.insert(hidden_edges.end(), e);
544  e.data->hidden = true;
545  ++hidden_edges_count;
546  }
547 
548  post_hide_edge_handler (e);
549 }

556 void graph::restore_edge (edge e)
557 {
558  assert (e.data->owner == this);
559  assert (e.data->owner == this);
560 
561  pre_restore_edge_handler (e);
562 
563  if (e.is_hidden()) {
564  //
565  // remove e from hidden edges list
566  //
567  hidden_edges.erase (e.data->pos);
568  --hidden_edges_count;
569 
570  //
571  // for each source of e insert e in its list of out-edges and store
572  // the position in e's list of positions
573  //
574  nodes_t::iterator it;
575  nodes_t::iterator end = e.data->nodes[0].end();
576 
577  for (it = e.data->nodes[0].begin (); it != end; ++it)
578  {
579  edges_t& adj = it->data->edges[1];
580  e.data->adj_pos[0].push_back(adj.insert(adj.begin(), e));
581  }
582 
583  //
584  // for each target of e insert e in its list of in-edges and store
585  // the pos
586  //
587  end = e.data->nodes[1].end();
588 
589  for (it = e.data->nodes[1].begin (); it != end; ++it)
590  {
591  edges_t& adj = it->data->edges[0];
592  e.data->adj_pos[1].push_back(adj.insert(adj.begin(), e));
593  }
594 
595  e.data->pos = edges.insert(edges.end(), e);
596  e.data->hidden = false;
597  }
598 
599  post_restore_edge_handler (e);
600 }

603 // Hide
604 // If an node is already hidden (this really happens :-), it will not be
605 // hidden for the second time
606 // Note: also all adjacent edges will be hidden
607 //--------------------------------------------------------------------------
608 
609 edges_t graph::hide_node(node n)
610 {
611  assert (n.data->owner == this);
612 
613  pre_hide_node_handler (n);
614  edges_t implicitly_hidden_edges;
615 
616  if (!n.is_hidden()){
617  // hide all connected egdes
618  for (int i = 0; i <= 1; ++i)
619  {
620  edges_t::iterator end = n.data->edges[i].end();
621  edges_t::iterator edge = n.data->edges[i].begin();
622  while (edge != end)
623  {
624  implicitly_hidden_edges.push_back(*edge);
625  hide_edge(*edge);
626  edge = n.data->edges[i].begin();
627  }
628  }
629 
630  // hide node
631  hidden_nodes.push_back(n);
632  nodes.erase(n.data->pos);
633  n.data->hidden = true;
634  ++hidden_nodes_count;
635  }
636 
637  post_hide_node_handler (n);
638 
639  return implicitly_hidden_edges;
640 }
