50 int maxflow_ff::check(graph& G)
51 {
52  if (!set_vars_executed)
53  {
54  return(GTL_ERROR);
55  }
56  graph::edge_iterator edge_it = G.edges_begin();
57  graph::edge_iterator edges_end = G.edges_end();
58  while (edge_it != edges_end)
59  {
60  if (edge_capacity[*edge_it] < 0)
61  {
62  return(GTL_ERROR);
63  }
64  ++edge_it;
65  }
66  // G.is_acyclic may be false
67  if ((G.number_of_nodes() <= 1) || (!G.is_connected()) || (G.is_undirected()))
68  {
69  return(GTL_ERROR);
70  }
71  if (artif_source_target)
72  {
73  bool source_found = false;
74  bool target_found = false;
75  graph::node_iterator node_it = G.nodes_begin();
76  graph::node_iterator nodes_end = G.nodes_end();
77  while (node_it != nodes_end)
78  {
79  if (node_it->indeg() == 0)
80  {
81  source_found = true;
82  }
83  if (node_it->outdeg() == 0)
84  {
85  target_found = true;
86  }
87  ++node_it;
88  }
89  if (!(source_found && target_found))
90  {
91  return(GTL_ERROR);
92  }
93  }
94  else
95  {
96  if (net_source == net_target)
97  {
98  return(GTL_ERROR);
99  }
100  }
101  return(GTL_OK); // ok
102 }

149 void maxflow_ff::create_artif_source_target(graph& G)
150 {
151  net_source = G.new_node();
152  net_target = G.new_node();
153  edge e;
154  graph::node_iterator node_it = G.nodes_begin();
155  graph::node_iterator nodes_end = G.nodes_end();
156  while (node_it != nodes_end)
157  {
158  if (*node_it != net_source && node_it->indeg() == 0)
159  {
160  e = G.new_edge(net_source, *node_it);
161  edge_capacity[e] = 1.0; // 1.0 prevents e from hiding
162  node::out_edges_iterator out_edge_it = node_it->out_edges_begin();
163  node::out_edges_iterator out_edges_end = node_it->out_edges_end();
164  while (out_edge_it != out_edges_end)
165  {
166  edge_capacity[e] += edge_capacity[*out_edge_it];
167  ++out_edge_it;
168  }
169  }
170  if (*node_it != net_target && node_it->outdeg() == 0)
171  {
172  e = G.new_edge(*node_it, net_target);
173  edge_capacity[e] = 1.0; // 1.0 prevents e from hiding
174  node::in_edges_iterator in_edge_it = node_it->in_edges_begin();
175  node::in_edges_iterator in_edges_end = node_it->in_edges_end();
176  while (in_edge_it != in_edges_end)
177  {
178  edge_capacity[e] += edge_capacity[*in_edge_it];
179  ++in_edge_it;
180  }
181  }
182  ++node_it;
183  }
184 }

196 void maxflow_ff::comp_single_flow(graph& G, node_map<edge>& last_edge)
197 {
198  double min_value = extra_charge(last_edge);
199 
200  node cur_node = net_target;
201  do
202  {
203  if (edge_org[last_edge[cur_node]]) // shortest path runs over a org. edge
204  {
205  if (!back_edge_exists[last_edge[cur_node]]) // create back edge
206  {
207  create_back_edge(G, last_edge[cur_node]);
208  }
209  edge_max_flow[last_edge[cur_node]] += min_value;
210  G.restore_edge(back_edge[last_edge[cur_node]]);
211  edge_capacity[back_edge[last_edge[cur_node]]] += min_value;
212  }
213  else // shortest path runs over a inserted back edge
214  {
215  edge oe = back_edge[last_edge[cur_node]];
216  G.restore_edge(oe);
217  edge_max_flow[oe] -= min_value;
218  edge_capacity[last_edge[cur_node]] -= min_value;
219  }
220  if (edge_capacity[last_edge[cur_node]] <= edge_max_flow[last_edge[cur_node]])
221  {
222  G.hide_edge(last_edge[cur_node]);
223  }
224  cur_node = last_edge[cur_node].source();
225  }
226  while (cur_node != net_source);
227 }

248 int maxflow_ff::comp_sp(const graph& /*G*/, std::queue<node>& next_nodes,
249  node_map<bool>& visited, node_map<edge>& last_edge)
250 {
251  node cur_node;
252 
253  while (!next_nodes.empty())
254  {
255  cur_node = next_nodes.front();
256  next_nodes.pop();
257 
258  node::out_edges_iterator out_edge_it = cur_node.out_edges_begin();
259  node::out_edges_iterator out_edges_end = cur_node.out_edges_end();
260  while (out_edge_it != out_edges_end)
261  {
262  node next = out_edge_it->target();
263  if (!visited[next])
264  {
265  last_edge[next] = *out_edge_it;
266  if (next == net_target)
267  {
268  return(SP_FOUND);
269  }
270  else
271  {
272  next_nodes.push(next);
273  visited[next] = true;
274  }
275  }
276  ++out_edge_it;
277  }
278  }
279  return(NO_SP_FOUND);
280 }

283 double maxflow_ff::extra_charge(const node_map<edge>& last_edge) const
284 {
285  node cur_node = net_target;
286  double min_value =
287  edge_capacity[last_edge[cur_node]] - edge_max_flow[last_edge[cur_node]];
288  double cur_capacity;
289 
290  do
291  {
292  cur_capacity =
293  edge_capacity[last_edge[cur_node]] - edge_max_flow[last_edge[cur_node]];
294 
295  if (cur_capacity < min_value) min_value = cur_capacity;
296  cur_node = last_edge[cur_node].source();
297  }
298  while (cur_node != net_source);
299  return(min_value);
300 }
