50 int maxflow_pp::check(graph& G)
51 {
52  if (!set_vars_executed)
53  {
54  return(GTL_ERROR);
55  }
56  graph::edge_iterator edge_it = G.edges_begin();
57  graph::edge_iterator edges_end = G.edges_end();
58  while (edge_it != edges_end)
59  {
60  if (edge_capacity[*edge_it] < 0)
61  {
62  return(GTL_ERROR);
63  }
64  ++edge_it;
65  }
66  // G.is_acyclic may be false
67  if ((G.number_of_nodes() <= 1) || (!G.is_connected()) || (G.is_undirected()))
68  {
69  return(GTL_ERROR);
70  }
71  if (artif_source_target)
72  {
73  bool source_found = false;
74  bool target_found = false;
75  graph::node_iterator node_it = G.nodes_begin();
76  graph::node_iterator nodes_end = G.nodes_end();
77  while (node_it != nodes_end)
78  {
79  if (node_it->indeg() == 0)
80  {
81  source_found = true;
82  }
83  if (node_it->outdeg() == 0)
84  {
85  target_found = true;
86  }
87  ++node_it;
88  }
89  if (!(source_found && target_found))
90  {
91  return(GTL_ERROR);
92  }
93  }
94  else
95  {
96  if (net_source == net_target)
97  {
98  return(GTL_ERROR);
99  }
100  }
101  return(GTL_OK); // ok
102 }

105 int maxflow_pp::run(graph& G)
106 {
107  // init
108  if (artif_source_target)
109  {
110  create_artif_source_target(G);
111  }
112  prepare_run(G);
113 
114  double flow_value = 0;
115  node min_tp_node;
116  while(leveling(G) == TARGET_FROM_SOURCE_REACHABLE)
117  {
118  hide_unreachable_nodes(G);
119  min_throughput_node(G, min_tp_node, flow_value);
120  push(G, min_tp_node, flow_value);
121  pull(G, min_tp_node, flow_value);
122  comp_rem_net(G);
123  }
124 
125  restore_graph(G);
126  return(GTL_OK);
127 }

153 void maxflow_pp::create_artif_source_target(graph& G)
154 {
155  net_source = G.new_node();
156  net_target = G.new_node();
157  edge e;
158  graph::node_iterator node_it = G.nodes_begin();
159  graph::node_iterator nodes_end = G.nodes_end();
160  while (node_it != nodes_end)
161  {
162  if (*node_it != net_source && node_it->indeg() == 0)
163  {
164  e = G.new_edge(net_source, *node_it);
165  edge_capacity[e] = 1.0; // 1.0 prevents e from hiding
166  node::out_edges_iterator out_edge_it = node_it->out_edges_begin();
167  node::out_edges_iterator out_edges_end = node_it->out_edges_end();
168  while (out_edge_it != out_edges_end)
169  {
170  edge_capacity[e] += edge_capacity[*out_edge_it];
171  ++out_edge_it;
172  }
173  }
174  if (*node_it != net_target && node_it->outdeg() == 0)
175  {
176  e = G.new_edge(*node_it, net_target);
177  edge_capacity[e] = 1.0; // 1.0 prevents e from hiding
178  node::in_edges_iterator in_edge_it = node_it->in_edges_begin();
179  node::in_edges_iterator in_edges_end = node_it->in_edges_end();
180  while (in_edge_it != in_edges_end)
181  {
182  edge_capacity[e] += edge_capacity[*in_edge_it];
183  ++in_edge_it;
184  }
185  }
186  ++node_it;
187  }
188 }

204 int maxflow_pp::leveling(graph& G)
205 {
206  bool source_target_con = false;
207  node_map<int> level(G, -1); // -1 means no level yet!
208  std::queue<node> next_nodes;
209  next_nodes.push(net_source);
210  level[net_source] = 0;
211  node cur_node;
212  while (!next_nodes.empty())
213  {
214  cur_node = next_nodes.front();
215  next_nodes.pop();
216  node::out_edges_iterator out_edge_it = cur_node.out_edges_begin();
217  node::out_edges_iterator out_edges_end = cur_node.out_edges_end();
218  while (out_edge_it != out_edges_end)
219  {
220  if (level[out_edge_it->target()] == -1)
221  {
222  if (out_edge_it->target() == net_target)
223  {
224  source_target_con = true;
225  }
226  level[out_edge_it->target()] = level[cur_node] + 1;
227  next_nodes.push(out_edge_it->target());
228  ++out_edge_it;
229  }
230  else if (level[out_edge_it->target()] <= level[cur_node])
231  {
232  node::out_edges_iterator temp_it = out_edge_it;
233  ++out_edge_it;
234  temp_unvisible_edges.push_back(*temp_it);
235  G.hide_edge(*temp_it);
236  }
237  else
238  {
239  ++out_edge_it;
240  }
241  }
242  }
243  if (source_target_con)
244  {
245  return(TARGET_FROM_SOURCE_REACHABLE);
246  }
247  else
248  {
249  return(TARGET_FROM_SOURCE_NOT_REACHABLE);
250  }
251 }

254 void maxflow_pp::hide_unreachable_nodes(graph& G)
255 {
256  node_map<bool> reachable_from_net_source(G, false);
257  node_map<bool> reachable_from_net_target(G, false);
258  std::queue<node> next_nodes;
259  node cur_node;
260 
261  next_nodes.push(net_source);
262  reachable_from_net_source[net_source] = true;
263  while (!next_nodes.empty())
264  {
265  cur_node = next_nodes.front();
266  next_nodes.pop();
267  node::out_edges_iterator out_edge_it = cur_node.out_edges_begin();
268  node::out_edges_iterator out_edges_end = cur_node.out_edges_end();
269  while (out_edge_it != out_edges_end)
270  {
271  node next = out_edge_it->target();
272  if (!reachable_from_net_source[next])
273  {
274  next_nodes.push(next);
275  reachable_from_net_source[next] = true;
276  }
277  ++out_edge_it;
278  }
279  }
280 
281  next_nodes.push(net_target);
282  reachable_from_net_target[net_target] = true;
283  while (!next_nodes.empty())
284  {
285  cur_node = next_nodes.front();
286  next_nodes.pop();
287  node::in_edges_iterator in_edge_it = cur_node.in_edges_begin();
288  node::in_edges_iterator in_edges_end = cur_node.in_edges_end();
289  while (in_edge_it != in_edges_end)
290  {
291  node next = in_edge_it->source();
292  if (!reachable_from_net_target[next])
293  {
294  next_nodes.push(next);
295  reachable_from_net_target[next] = true;
296  }
297  ++in_edge_it;
298  }
299  }
300 
301  graph::node_iterator node_it = G.nodes_begin();
302  graph::node_iterator nodes_end = G.nodes_end();
303  while (node_it != nodes_end)
304  {
305  if ((!reachable_from_net_source[*node_it]) ||
306  (!reachable_from_net_target[*node_it]))
307  {
308  graph::node_iterator temp_it = node_it;
309  ++node_it;
310  temp_unvisible_nodes.push_back(*temp_it);
311  store_temp_unvisible_edges(*temp_it);
312  G.hide_node(*temp_it);
313  }
314  else
315  {
316  ++node_it;
317  }
318  }
319 }

363 double maxflow_pp::comp_min_throughput(const node cur_node) const
364 {
365  double in_flow = 0.0;
366  double out_flow = 0.0;
367  node::in_edges_iterator in_it = cur_node.in_edges_begin();
368  node::in_edges_iterator in_edges_end = cur_node.in_edges_end();
369  while (in_it != in_edges_end)
370  {
371  in_flow += edge_capacity[*in_it] - edge_max_flow[*in_it];
372  ++in_it;
373  }
374  node::out_edges_iterator out_it = cur_node.out_edges_begin();
375  node::out_edges_iterator out_edges_end = cur_node.out_edges_end();
376  while (out_it != out_edges_end)
377  {
378  out_flow += edge_capacity[*out_it] - edge_max_flow[*out_it];
379  ++out_it;
380  }
381  if (cur_node == net_source)
382  {
383  return(out_flow);
384  }
385  if (cur_node == net_target)
386  {
387  return(in_flow);
388  }
389  return(in_flow < out_flow ? in_flow : out_flow);
390 }

463 void maxflow_pp::push(graph& G, const node& start_node, const double flow_value)
464 {
465  node_map<edge> last_edge;
466  double cur_flow = flow_value;
467  double min_value = 0.0;
468 
469  if (start_node == net_target)
470  {
471  return; // no push necessary
472  }
473  do
474  {
475  get_sp_ahead(G, start_node, last_edge);
476  min_value = extra_charge_ahead(start_node, last_edge);
477  if (min_value > cur_flow)
478  {
479  min_value = cur_flow;
480  }
481  node cur_node = net_target;
482  do
483  {
484  if (edge_org[last_edge[cur_node]])
485  {
486  edge_max_flow[last_edge[cur_node]] += min_value;
487  if (back_edge_exists[last_edge[cur_node]])
488  {
489  flow_update[back_edge[last_edge[cur_node]]] += min_value;
490  }
491  }
492  else
493  {
494  edge_capacity[last_edge[cur_node]] -= min_value;
495  flow_update[back_edge[last_edge[cur_node]]] += min_value;
496  }
497  if (edge_capacity[last_edge[cur_node]] <=
498  edge_max_flow[last_edge[cur_node]])
499  {
500  full_edges.push_back(last_edge[cur_node]);
501  G.hide_edge(last_edge[cur_node]);
502  }
503  cur_node = last_edge[cur_node].source();
504  }
505  while (cur_node != start_node);
506  cur_flow -= min_value;
507  if (cur_flow < 1e-015) // quite hacky ;-)
508  {
509  cur_flow = 0.0; // to avoid rounding errors
510  }
511  } while (cur_flow > 0.0);
512 }

567 void maxflow_pp::comp_rem_net(graph& G)
568 {
569  // update back_edges
570  graph::edge_iterator edge_it = G.edges_begin();
571  graph::edge_iterator edges_end = G.edges_end();
572  while (edge_it != edges_end)
573  {
574  single_edge_update(G, *edge_it);
575  ++edge_it;
576  }
577  edges_t::iterator list_it = full_edges.begin();
578  edges_t::iterator list_end = full_edges.end();
579  while (list_it != list_end)
580  {
581  G.restore_edge(*list_it);
582  if (flow_update[*list_it] > 0.0)
583  {
584  single_edge_update(G, *list_it);
585  edges_t::iterator temp_it = list_it;
586  ++list_it;
587  full_edges.erase(temp_it); // now it's visible again
588  }
589  else
590  {
591  if (!back_edge_exists[*list_it])
592  {
593  create_back_edge(G, *list_it);
594  edge_capacity[back_edge[*list_it]] = edge_max_flow[*list_it];
595  }
596  G.hide_edge(*list_it);
597  ++list_it;
598  }
599  }
600 
601 
602  // make hidden levels visible again
603  nodes_t::iterator temp_un_node_it = temp_unvisible_nodes.begin();
604  nodes_t::iterator temp_un_nodes_end = temp_unvisible_nodes.end();
605  while (temp_un_node_it != temp_un_nodes_end)
606  {
607  G.restore_node(*temp_un_node_it);
608  ++temp_un_node_it;
609  }
610  edges_t::iterator temp_un_edge_it = temp_unvisible_edges.begin();
611  edges_t::iterator temp_un_edges_end = temp_unvisible_edges.end();
612  while (temp_un_edge_it != temp_un_edges_end)
613  {
614  G.restore_edge(*temp_un_edge_it);
615  if (flow_update[*temp_un_edge_it] > 0.0)
616  {
617  single_edge_update(G, *temp_un_edge_it);
618  }
619  ++temp_un_edge_it;
620  }
621  temp_unvisible_nodes.clear();
622  temp_unvisible_edges.clear();
623 }

626 void maxflow_pp::single_edge_update(graph& G, edge cur_edge)
627 {
628  if(edge_org[cur_edge])
629  {
630  edge_max_flow[cur_edge] -= flow_update[cur_edge];
631  flow_update[cur_edge] = 0.0;
632  if (!back_edge_exists[cur_edge])
633  {
634  if (edge_max_flow[cur_edge] > 0.0)
635  {
636  create_back_edge(G, cur_edge);
637  edge_capacity[back_edge[cur_edge]] = edge_max_flow[cur_edge];
638  }
639  }
640  }
641  else
642  {
643  edge_capacity[cur_edge] += flow_update[cur_edge];
644  flow_update[cur_edge] = 0.0;
645  }
646 }

651 {

654  - edge_max_flow[last_edge[cur_node]];
655  double cur_capacity;
656

659  cur_capacity = edge_capacity[last_edge[cur_node]]
660  - edge_max_flow[last_edge[cur_node]];
661  if (cur_capacity < min_value) min_value = cur_capacity;
662  cur_node = last_edge[cur_node].source();
663  }
664  while (cur_node != start_node);
665  return(min_value);

703 void maxflow_pp::comp_max_flow(const graph& /*G*/)
704 {
705  max_graph_flow = 0.0;
706
