49 int maxflow_sap::check(graph& G)
50 {
51  if (!set_vars_executed)
52  {
53  return(GTL_ERROR);
54  }
55  graph::edge_iterator edge_it = G.edges_begin();
56  graph::edge_iterator edges_end = G.edges_end();
57  while (edge_it != edges_end)
58  {
59  if (edge_capacity[*edge_it] < 0)
60  {
61  return(GTL_ERROR);
62  }
63  ++edge_it;
64  }
65  // G.is_acyclic may be false
66  if ((G.number_of_nodes() <= 1) || (!G.is_connected())
67  || (G.is_undirected()))
68  {
69  return(GTL_ERROR);
70  }
71  if (artif_source_target)
72  {
73  bool source_found = false;
74  bool target_found = false;
75  graph::node_iterator node_it = G.nodes_begin();
76  graph::node_iterator nodes_end = G.nodes_end();
77  while (node_it != nodes_end)
78  {
79  if (node_it->indeg() == 0)
80  {
81  source_found = true;
82  }
83  if (node_it->outdeg() == 0)
84  {
85  target_found = true;
86  }
87  ++node_it;
88  }
89  if (!(source_found && target_found))
90  {
91  return(GTL_ERROR);
92  }
93  }
94  else
95  {
96  if (net_source == net_target)
97  {
98  return(GTL_ERROR);
99  }
100  }
101  return(GTL_OK); // everything ok
102 }

105 int maxflow_sap::run(graph& G)
106 {
107  // init
108  if (artif_source_target)
109  {
110  create_artif_source_target(G);
111  }
112  bool go_on = true;
113  node_map<edge> last_edge(G);
114  node cur_node;
115  int number_of_nodes = G.number_of_nodes();
116  std::vector<int> numb(number_of_nodes, 0);
117  prepare_run(G);
118 
119  comp_dist_labels(G, numb);
120  cur_node = net_source;
121 
122  while (go_on)
123  {
124  if (has_an_admissible_arc(cur_node))
125  {
126  advance(cur_node, last_edge);
127  if (cur_node == net_target)
128  {
129  augment(G, last_edge);
130  cur_node = net_source;
131  }
132  }
133  else // only inadmissible edges
134  {
135  go_on = retreat(number_of_nodes, cur_node, last_edge, numb);
136  }
137  }
138 
139  restore_graph(G);
140  return(GTL_OK);
141 }

169 void maxflow_sap::create_artif_source_target(graph& G)
170 {
171  net_source = G.new_node();
172  net_target = G.new_node();
173  edge e;
174  graph::node_iterator node_it = G.nodes_begin();
175  graph::node_iterator nodes_end = G.nodes_end();
176  while (node_it != nodes_end)
177  {
178  if (*node_it != net_source && node_it->indeg() == 0)
179  {
180  e = G.new_edge(net_source, *node_it);
181  edge_capacity[e] = 1.0; // 1.0 prevents e from hiding
182  node::out_edges_iterator out_edge_it =
183  node_it->out_edges_begin();
184  node::out_edges_iterator out_edges_end =
185  node_it->out_edges_end();
186  while (out_edge_it != out_edges_end)
187  {
188  edge_capacity[e] += edge_capacity[*out_edge_it];
189  ++out_edge_it;
190  }
191  }
192  if (*node_it != net_target && node_it->outdeg() == 0)
193  {
194  e = G.new_edge(*node_it, net_target);
195  edge_capacity[e] = 1.0; // 1.0 prevents e from hiding
196  node::in_edges_iterator in_edge_it =
197  node_it->in_edges_begin();
198  node::in_edges_iterator in_edges_end =
199  node_it->in_edges_end();
200  while (in_edge_it != in_edges_end)
201  {
202  edge_capacity[e] += edge_capacity[*in_edge_it];
203  ++in_edge_it;
204  }
205  }
206  ++node_it;
207  }
208 }

220 void maxflow_sap::comp_dist_labels(const graph& G, std::vector<int>& numb)
221 {
222  std::queue<node> next_nodes;
223  node_map<bool> visited(G, false);
224  next_nodes.push(net_target);
225  visited[net_target] = true;
226  dist_label[net_target] = 0;
227  numb[0] = 1; // only one sink
228  node cur_node;
229 
230  while (!next_nodes.empty())
231  {
232  cur_node = next_nodes.front();
233  next_nodes.pop();
234  node::in_edges_iterator in_edge_it = cur_node.in_edges_begin();
235  node::in_edges_iterator in_edges_end = cur_node.in_edges_end();
236  while (in_edge_it != in_edges_end)
237  {
238  node next = in_edge_it->source();
239  if (!visited[next])
240  {
241  next_nodes.push(next);
242  visited[next] = true;
243  dist_label[next] = dist_label[cur_node] + 1;
244  ++numb[dist_label[next]];
245  }
246  ++in_edge_it;
247  }
248  }
249 }

284 void maxflow_sap::augment(graph& G, const node_map<edge>& last_edge)
285 {
286  double additional_flow = free_capacity(last_edge);
287  node cur_node = net_target;
288 
289  do
290  {
291  if (edge_org[last_edge[cur_node]])
292  // shortest path runs over a org. edge
293  {
294  if (!back_edge_exists[last_edge[cur_node]]) // create back edge
295  {
296  create_back_edge(G, last_edge[cur_node]);
297  }
298  edge_max_flow[last_edge[cur_node]] += additional_flow;
299  G.restore_edge(back_edge[last_edge[cur_node]]);
300  edge_capacity[back_edge[last_edge[cur_node]]] +=
301  additional_flow;
302  }
303  else // shortest path runs over a inserted back edge
304  {
305  edge oe = back_edge[last_edge[cur_node]];
306  G.restore_edge(oe);
307  edge_max_flow[oe] -= additional_flow;
308  edge_capacity[last_edge[cur_node]] -= additional_flow;
309  }
310  if (edge_capacity[last_edge[cur_node]] <=
311  edge_max_flow[last_edge[cur_node]])
312  {
313  G.hide_edge(last_edge[cur_node]);
314  }
315  cur_node = last_edge[cur_node].source();
316  }
317  while (cur_node != net_source);
318 }

321 bool maxflow_sap::retreat(const int number_of_nodes,
322  node& cur_node,
323  const node_map<edge>& last_edge,
324  std::vector<int>& numb)
325 {
326  --numb[dist_label[cur_node]];
327  if (numb[dist_label[cur_node]] == 0)
328  {
329  return false;
330  }
331  else
332  {
333  dist_label[cur_node] =
334  min_neighbour_label(number_of_nodes, cur_node) + 1;
335  ++numb[dist_label[cur_node]];
336  if (cur_node != net_source)
337  {
338  cur_node = last_edge[cur_node].source();
339  }
340  return true;
341  }
342 }

364 double maxflow_sap::free_capacity(const node_map<edge>& last_edge) const
365 {
366  node cur_node = net_target;
367  double min_value =
368  edge_capacity[last_edge[cur_node]] -
369  edge_max_flow[last_edge[cur_node]];
370  double cur_capacity;
371 
372  do
373  {
374  cur_capacity =
375  edge_capacity[last_edge[cur_node]] -
376  edge_max_flow[last_edge[cur_node]];
377 
378  if (cur_capacity < min_value)
379  {
380  min_value = cur_capacity;
381  }
382  cur_node = last_edge[cur_node].source();
383  }
384  while (cur_node != net_source);
385  return(min_value);
386 }
