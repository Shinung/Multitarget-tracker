138 int mwbmatching::augment(graph& G, node a)
139 {
140  // Initialise
141  pred.init(G, -1);
142  pq = fh_alloc(G.number_of_nodes());
143 
144  dist[a] = 0;
145  node best_node_in_A = a;
146  long minA = pot[a];
147  long delta;
148 
149  std::stack<node, std::vector<node> > RA;
150  RA.push(a);
151  std::stack<node, std::vector<node> > RB;
152 
153  node a1 = a;
154  edge e;
155 
156  // Relax
157  forall_adj_edges (e, a1)
158  {
159  const node& b = e.target_();
160  long db = dist[a1] + (pot[a1] + pot[b] - edge_weight[e]);
161  if (pred[b] == -1)
162  {
163  dist[b] = db;
164  pred[b] = e.id();
165  RB.push(b);
166 
167  fh_insert (pq, b.id(), db);
168  }
169  else
170  {
171  if (db < dist[b])
172  {
173  dist[b] = db;
174  pred[b] = e.id();
175 
176  fh_decrease_key (pq, b.id(), db);
177  }
178  }
179  }
180 
181  for (;;)
182  {
183  // Find node with minimum distance db
184  int node_id;
185  long db = 0;
186  if (pq->n == 0)
187  {
188  node_id = -1;
189  }
190  else
191  {
192  node_id = fh_delete_min (pq);
193  db = dist[node_from_id[node_id]];
194  }
195 
196  if (node_id == -1 || db >= minA)
197  {
198  delta = minA;
199  // augmentation by best node in A
200  augment_path_to (G, best_node_in_A);
201  free[a] = false;
202  free[best_node_in_A] = true;
203  break;
204  }
205  else
206  {
207  node b = node_from_id[node_id];
208  if (free[b])
209  {
210  delta = db;
211  // augmentation by path to b, so a and b are now matched
212  augment_path_to (G, b);
213  free[a] = false;
214  free[b] = false;
215  break;
216  }
217  else
218  {
219  // continue shortest path computation
220  edge e = (*b.adj_edges_begin());
221  const node& a1 = e.target_();
222  pred[a1] = e.id();
223  RA.push(a1);
224  dist[a1] = db;
225 
226  if (db + pot[a1] < minA)
227  {
228  best_node_in_A = a1;
229  minA = db + pot[a1];
230  }
231 
232  // Relax
233  forall_adj_edges (e, a1)
234  {
235  const node& b = e.target_();
236  long db = dist[a1] + (pot[a1] + pot[b] - edge_weight[e]);
237  if (pred[b] == -1)
238  {
239  dist[b] = db;
240  pred[b] = e.id();
241  RB.push(b);
242 
243  fh_insert (pq, b.id(), db);
244  }
245  else
246  {
247  if (db < dist[b])
248  {
249  dist[b] = db;
250  pred[b] = e.id();
251 
252  fh_decrease_key (pq, b.id(), db);
253  }
254  }
255  }
256 
257  }
258  }
259  }
260 
261 
262  while (!RA.empty())
263  {
264  node a = RA.top();
265  RA.pop();
266  pred[a] = -1;
267  long pot_change = delta - dist[a];
268  if (pot_change <= 0) continue;
269  pot[a] = pot[a] - pot_change;
270  }
271  while (!RB.empty())
272  {
273  node b = RB.top();
274  RB.pop();
275  pred[b] = -1;
276 
277  long pot_change = delta - dist[b];
278  if (pot_change <= 0) continue;
279  pot[b] = pot[b] + pot_change;
280  }
281 
282  // Clean up
283  fh_free(pq);
284 
285  return 0;
