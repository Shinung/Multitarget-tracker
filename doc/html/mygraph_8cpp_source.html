333 double MyGraph::node_cliqueishness (node &n)
334 {
335  double c = 1.0;
336 
337  int numneighbours = n.degree();
338  int possconnections = numneighbours * (numneighbours - 1) / 2;
339  int actualconnections = 0;
340  if (possconnections > 0)
341  {
342  // Build list of all nodes adjacent to n (n's neighbours)
343  nodes_t neighbours;
344 
345  node::adj_nodes_iterator nit = n.adj_nodes_begin ();
346  node::adj_nodes_iterator nend = n.adj_nodes_end ();
347  while (nit != nend)
348  {
349  node ne = (*nit);
350  neighbours.push_back (ne);
351  nit++;
352  }
353 
354  // Count number of edges between neighbours
355  nodes_t::iterator i = neighbours.begin();
356  nodes_t::iterator iend = neighbours.end();
357  while (i != iend)
358  {
359  nodes_t::iterator j = i;
360  j++;
361 
362  while (j != iend)
363  {
364  if (edge_exists (*i, *j))
365  actualconnections++;
366  j++;
367  }
368  i++;
369  }
370  c = (double) actualconnections / (double) possconnections;
371  }
372  return c;
373 }
