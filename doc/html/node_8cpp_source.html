170 node::adj_edges_iterator::adj_edges_iterator(node n, bool start)
171 {
172  // iterators that are used everytime
173  last_edge[0] = n.out_edges_end();
174  last_edge[1] = n.in_edges_end();
175  directed = n.is_directed();
176  if (!directed)
177  {
178  begin_edge[0] = n.out_edges_begin();
179  begin_edge[1] = n.in_edges_begin();
180  }
181 
182  // set at start or end
183  if (start)
184  {
185  inout = 0;
186  akt_edge[0] = n.out_edges_begin();
187  if (!directed)
188  {
189  akt_edge[1] = n.in_edges_begin();
190  if (akt_edge[0] == last_edge[0])
191  inout = 1;
192  }
193  }
194  else
195  {
196  inout = directed ? 0 : 1;
197  akt_edge[0] = n.out_edges_end();
198  if (!directed)
199  akt_edge[1] = n.in_edges_end();
200  }
201 }
