47 bool planarity::run_on_biconnected (graph& G, planar_embedding& em)
48 {
49 
50  if (G.number_of_edges() == 0) return algorithm::GTL_OK;
51 
52  st_number st_;
53 
54  //
55  // The graph may have self loops. Make sure that we
56  // choose a normal edge for st.
57  //
58 
59  graph::edge_iterator
60  edge_it = G.edges_begin(),
61  edge_end = G.edges_end();
62 
63  edge st;
64 
65  while (edge_it != edge_end) {
66  if (edge_it->source() != edge_it->target()) {
67  st = *edge_it;
68  break;
69  }
70  ++edge_it;
71  }
72 
73  //
74  // G has only selfloops
75  //
76 
77  if (st == edge()) {
78  if (emp) {
79  em.init (G);
80  edge_it = G.edges_begin();
81  edge_end = G.edges_end();
82 
83  for (;edge_it != edge_end; ++edge_it) {
84  em.self.push_back (*edge_it);
85  }
86  }
87 
88  return algorithm::GTL_OK;
89  }
90 
91  st_.st_edge (st);
92  st_.s_node (st.source());
93  int res = st_.check(G);
94  assert (res == algorithm::GTL_OK);
95  res = st_.run(G);
96  assert (res == algorithm::GTL_OK);
97  int size = G.number_of_nodes();
98 
99  if (emp) {
100  em.init (G);
101  }
102 
103  std::list<pq_leaf*> neighbors;
104  st_number::iterator st_it = st_.begin();
105  node curr = *st_it;
106  node::out_edges_iterator o_it = curr.out_edges_begin();
107  node::out_edges_iterator o_end = curr.out_edges_end();
108  node::in_edges_iterator i_it = curr.in_edges_begin();
109  node::in_edges_iterator i_end = curr.in_edges_end();
110  edges_t self_loops;
111  node opp;
112  node_map<int> visited_from (G, 0);
113  pq_leaf* tmp_leaf;
114  std::vector< std::list<pq_leaf*> > leaves(size);
115 
116  for (; o_it != o_end; ++o_it) {
117  opp = curr.opposite (*o_it);
118 
119  if (opp != curr) {
120  if (visited_from[opp] == st_[curr] && emp) {
121  em.multi.push_back (*o_it);
122  } else {
123  visited_from[opp] = st_[curr];
124  tmp_leaf = new pq_leaf (st_[opp], st_[curr], *o_it, opp);
125  leaves[st_[opp]-1].push_back (tmp_leaf);
126  neighbors.push_back (tmp_leaf);
127  }
128 
129  } else if (emp) {
130  em.self.push_back (*o_it);
131  }
132  }
133 
134  for (; i_it != i_end; ++i_it) {
135  opp = curr.opposite (*i_it);
136 
137  if (opp != curr) {
138  if (visited_from[opp] == st_[curr] && emp) {
139  em.multi.push_back (*i_it);
140  } else {
141  visited_from[opp] = st_[curr];
142  tmp_leaf = new pq_leaf (st_[opp], st_[curr], *i_it, opp);
143  leaves[st_[opp]-1].push_back (tmp_leaf);
144  neighbors.push_back (tmp_leaf);
145  }
146  }
147  }
148 
149  node_map<std::list<direction_indicator> > dirs;
150 
151  //
152  // There is a problem with node/edge maps of iterators with Visual C++
153  // which I don´t fully understand at the moment. Fortunatly the init for the
154  // maps below is only needed to allocate memory, which is done anyway, when
155  // values are assigned to it.
156  //
157 
158 #ifndef __GTL_MSVCC
159  dirs.init (G);
160 #endif
161  pq_tree PQ (st_[curr], curr, neighbors);
162  neighbors.erase (neighbors.begin(), neighbors.end());
163  ++st_it;
164  curr = *st_it;
165 
166  while (st_[curr] < size) {
167 
168 #ifdef _DEBUG
169  char filename[10] = "out";
170  char buffer[12];
171 #ifdef __GTL_MSVCC
172  _snprintf (buffer, 12, "%s%d.gml", filename, st_[curr]);
173 #else
174  snprintf (buffer, 12, "%s%d.gml", filename, st_[curr]);
175 #endif
176  std::ofstream os(buffer, std::ios::out | std::ios::trunc);
177  os << PQ << std::endl;
178  os.close();
179  bool ret_flag = PQ.integrity_check();
180  assert(ret_flag);
181 #endif
182 
183  if (!PQ.reduce (leaves[st_[curr]-1])) {
184 #ifdef _DEBUG
185  os.open("fail.gml", std::ios::out | std::ios::trunc);
186  os << PQ << std::endl;
187  os.close ();
188 #endif
189  if (kup) {
190  examine_obstruction (G, st_, curr,
191  PQ.get_fail(), PQ.is_fail_root(), em, dirs, &PQ);
192  }
193 
194  PQ.reset();
195  return false;
196  }
197 
198 
199  //
200  // It seems to be not very comfortable to use in and out iterators to
201  // go through the adjacency of a node. For graphs without selfloops this
202  // could be replaced by using adj_iterator, but if there are selfloops
203  // they will occur in both the list of outgoing and the one of incoming
204  // edges, and thus two times in the adjacency.
205  //
206 
207  o_it = curr.out_edges_begin();
208  o_end = curr.out_edges_end();
209  i_it = curr.in_edges_begin();
210  i_end = curr.in_edges_end();
211 
212  for (; o_it != o_end; ++o_it) {
213  opp = curr.opposite (*o_it);
214 
215  if (st_[opp] > st_[curr]) {
216  if (visited_from[opp] == st_[curr] && emp) {
217  em.multi.push_back (*o_it);
218  } else {
219  visited_from[opp] = st_[curr];
220  tmp_leaf = new pq_leaf (st_[opp], st_[curr], *o_it, opp);
221  leaves[st_[opp]-1].push_back (tmp_leaf);
222  neighbors.push_back (tmp_leaf);
223  }
224 
225  } else if (st_[opp] == st_[curr] && emp) {
226  em.self.push_back (*o_it);
227  }
228  }
229 
230  for (; i_it != i_end; ++i_it) {
231  opp = curr.opposite (*i_it);
232 
233  if (st_[opp] > st_[curr]) {
234  if (visited_from[opp] == st_[curr] && emp) {
235  em.multi.push_back (*i_it);
236  } else {
237  visited_from[opp] = st_[curr];
238  tmp_leaf = new pq_leaf (st_[opp], st_[curr], *i_it, opp);
239  leaves[st_[opp]-1].push_back (tmp_leaf);
240  neighbors.push_back (tmp_leaf);
241  }
242  }
243  }
244 
245  if (emp) {
246  PQ.replace_pert (st_[curr], curr, neighbors, &em, &(dirs[curr]));
247 #ifdef _DEBUG
248  GTL_debug::os() << "Embedding of " << st_[curr] << ":: ";
249  planar_embedding::iterator adit, adend;
250  for (adit = em.adj_edges_begin (curr), adend = em.adj_edges_end (curr); adit != adend; ++adit) {
251  GTL_debug::os() << "[" << st_[curr.opposite (*adit)] << "]";
252  }
253  GTL_debug::os() << std::endl;
254  GTL_debug::os() << "Direction Indicators for: " << st_[curr] << ":: ";
255  std::list<direction_indicator>::iterator dit, dend;
256 
257  for (dit = dirs[curr].begin(), dend = dirs[curr].end(); dit != dend; ++dit) {
258  GTL_debug::os() << "[";
259  if (dit->direction)
260  GTL_debug::os() << ">> " << dit->id << " >>";
261  else
262  GTL_debug::os() << "<< " << dit->id << " <<";
263  GTL_debug::os() << "]";
264  }
265  GTL_debug::os() << std::endl;
266 #endif
267 
268  } else {
269  PQ.replace_pert (st_[curr], curr, neighbors);
270  }
271 
272  PQ.reset();
273 
274 
275  neighbors.erase (neighbors.begin(), neighbors.end());
276  ++st_it;
277  curr = *st_it;
278  }
279 
280  if (emp) {
281  PQ.get_frontier (em, dirs[curr]);
282 
283 
284  //
285  // get self_loops for last node
286  //
287 
288  o_it = curr.out_edges_begin();
289  o_end = curr.out_edges_end();
290 
291  for (; o_it != o_end; ++o_it) {
292  if (o_it->target() == o_it->source()) {
293  em.self.push_back (*o_it);
294  }
295  }
296 
297  //
298  // some adjcacency list of the embedding obtained so far have to be
299  // turned.
300  //
301 
302  correct_embedding(em, st_, dirs);
303 
304  node_map<int> mark;
305  mark.init (G, 0);
306  node_map<symlist<edge>::iterator > upward_begin;
307  upward_begin.init (G);
308  node tmp;
309 
310  forall_nodes (tmp, G) {
311  upward_begin[tmp] = em.adjacency(tmp).begin();
312  }
313 
314  extend_embedding(curr, em, mark, upward_begin);
315  }
316 
317  return true;
318 }

321 int planarity::run (graph& G)
322 {
323  bool directed = false;
324 
325  if (G.is_directed()) {
326  G.make_undirected();
327  directed = true;
328  }
329 
330  biconnectivity biconn;
331 
332  if (bip) {
333  biconn.make_biconnected (true);
334  } else {
335  biconn.store_components (true);
336  }
337 
338  biconn.check (G);
339  biconn.run (G);
340 
341  if (emp) {
342  embedding.init (G);
343  }
344 
345  planar_embedding em;
346 
347  if (!biconn.is_biconnected() && !bip) {
348  biconnectivity::component_iterator c_it, c_end;
349 
350  for (c_it = biconn.components_begin(), c_end = biconn.components_end();
351  c_it != c_end; ++c_it) {
352 
353  switch_to_component (G, c_it);
354 
355 #ifdef _DEBUG
356  GTL_debug::os() << "Component is: " << std::endl;
357  GTL_debug::os() << G << std::endl;
358 #endif
359  if (!run_on_biconnected (G, em)) {
360  if (directed) {
361  G.make_directed();
362  }
363 
364  G.restore_graph();
365  planar = false;
366  return algorithm::GTL_OK;
367  }
368 
369  if (emp) {
370  add_to_embedding (G, em);
371  }
372  }
373 
374  G.restore_graph();
375 
376  } else {
377 
378  //
379  // G is already biconnected
380  //
381 
382  GTL_debug::debug_message ("graph is biconnected\n");
383 
384  if (!run_on_biconnected (G, embedding)) {
385  if (directed) {
386  G.make_directed();
387  }
388 
389  planar = false;
390  return algorithm::GTL_OK;
391  }
392  }
393 
394  if (bip) {
395  edges_t::iterator it, end;
396  it = biconn.additional_begin();
397  end = biconn.additional_end();
398 
399  for (; it != end; ++it) {
400 
401  if (emp) {
402  node s = it->source();
403  node t = it->target();
404  embedding.adj[s].erase (embedding.s_pos[*it]);
405  embedding.adj[t].erase (embedding.t_pos[*it]);
406  }
407 
408  G.del_edge (*it);
409  }
410  }
411 
412  if (directed) {
413  G.make_directed();
414  }
415 
416  planar = true;
417  return algorithm::GTL_OK;
418 }

453 void planarity::correct_embedding (
454  planar_embedding& em,
455  st_number& st_,
456  node_map<std::list<direction_indicator> >& dirs)
457 {
458  st_number::reverse_iterator it = st_.rbegin();
459  st_number::reverse_iterator end = st_.rend();
460  bool* turn = new bool[st_[*it]];
461 
462  for (int i = 0; i < st_[*it]; ++i) {
463  turn[i] = false;
464  }
465 
466  while (it != end) {
467  node curr = *it;
468 
469  if (turn[st_[curr] - 1]) {
470  em.adjacency(curr).reverse();
471  }
472 
473  std::list<direction_indicator>::iterator d_it = dirs[curr].begin();
474 
475  while (!dirs[curr].empty()) {
476 
477  if (d_it->direction && turn[st_[curr] - 1] ||
478  !d_it->direction && !turn[st_[curr] - 1]) {
479  turn[d_it->id - 1] = true;
480  }
481 
482  d_it = dirs[curr].erase (d_it);
483  }
484 
485  ++it;
486  }
487 
488  delete[] turn;
489 }

492 void planarity::extend_embedding (
493  node n,
494  planar_embedding& em,
495  node_map<int>& mark,
496  node_map<symlist<edge>::iterator >& upward_begin)
497 {
498  mark[n] = 1;
499 
500  symlist<edge>::iterator it = upward_begin[n];
501  symlist<edge>::iterator end = em.adjacency(n).end();
502  node other;
503 
504  for (; it != end; ++it) {
505  em.pos (n, *it) = it;
506  other = n.opposite (*it);
507  em.pos (other, *it) = em.push_front (other, *it);
508 
509  if (mark[other] == 0) {
510  extend_embedding (other, em, mark, upward_begin);
511  }
512  }

548 void planarity::examine_obstruction (graph& G,
549  st_number& st_,
550  node act,
551  pq_node* fail,
552  bool is_root,
553  planar_embedding& em,
554  node_map<std::list<direction_indicator> >& dirs,
555  pq_tree* PQ)
556 {
557  node_map<int> used (G, 0);
558  node_map<edge> to_father (G);
559 
560  //
561  // Create a dfs-tree of the so called bush form. This is basically a normal dfs
562  // applied to the induced subgraph of G consisting only of the nodes with st_number
563  // 1, ..., st_[act] - 1. The only difference is that edges are always directed from
564  // the lower numbered vertex to higher numbered one.
565  //
566 
567  dfs_bushform (st_.s_node(), used, st_, st_[act], to_father);
568 
569  if (fail->kind() == pq_node::Q_NODE) {
570 
571  //
572  // In case the reduction failed at a Q-Node we need to know the edges that
573  // form the boundary of the biconnected component, which this Q-Node represents.
574  // These can easily be obtained from the embedding we got so far.
575  //
576 
577  q_node* q_fail = fail->Q();
578 
579  pq_tree::sons_iterator s_it = q_fail->sons.begin();
580  pq_tree::sons_iterator s_end = q_fail->sons.end();
581  node greatest = fail->n;
582 
583  while (s_it != s_end) {
584  if ((*s_it)->kind() == pq_node::DIR) {
585  direction_indicator* dir = (*s_it)->D();
586  pq_tree::sons_iterator tmp = s_it;
587 
588  if (++tmp == ++(dir->pos)) {
589  dir->direction = true;
590  } else {
591  dir->direction = false;
592  }
593 
594  dirs[act].push_back (*dir);
595 
596  //
597  // chris 2/3/2008:
598  //
599  // To avoid a memory leak, it is not sufficient to erase it from the
600  // PQ-tree (-node). The direction indicator object also has to be
601  // deleted. Since it is then not a member of the pertinent subtree any
602  // more, it must not be cleared by PQ->reset(). The instance in the
603  // dirs node map is a clone!
604  //
605 
606  // s_it = q_fail->sons.erase (s_it);
607  s_it = PQ->remove_dir_ind(q_fail, s_it);
608  } else {
609  if (st_[(*s_it)->up] > st_[greatest]) {
610  greatest = (*s_it)->up;
611  }
612 
613  ++s_it;
614  }
615  }
616 
617  correct_embedding (em, st_, dirs);
618  node_map<int> mark;
619  mark.init (G, 0);
620  node_map<symlist<edge>::iterator > upward_begin;
621  upward_begin.init (G);
622  node tmp;
623 
624  em.adjacency(fail->n).erase (
625  em.adjacency(fail->n).begin(),
626  em.adjacency(fail->n).end());
627 
628  forall_nodes (tmp, G) {
629  upward_begin[tmp] = em.adjacency(tmp).begin();
630  }
631 
632  //
633  // chris 2/3/2008:
634  //
635  // With the code of MR 11/27/2001 the component of the failing Q-node is not found
636  // correctly.
637  //
638 
639  extend_embedding(greatest, em, mark, upward_begin);
640 
641  /*
642  //
643  // MR 11/27/2001:
644  //
645  // This is important! We restricted building the embedding to the nodes in
646  // the biconnected component which the Q-node fail refers to. But the st-number
647  // obtained for the whole graph restricted to these nodes will not be a st-numbering
648  // for this biconnected component.
649  //
650 
651  st_number::reverse_iterator st_it, st_end;
652 
653  for (st_it = st_.rbegin(), st_end = st_.rend();
654  st_it != st_end;
655  ++st_it)
656  {
657  if (mark[*st_it] == 0) {
658  extend_embedding (*st_it, em, mark, upward_begin);
659  }
660  }
661  */
662 #ifdef _DEBUG
663  GTL_debug::os() << "Embedding so far (st_numbered): " << std::endl;
664  em.write_st (GTL_debug::os(), st_);
665 #endif
666 
667  attachment_cycle (fail->n, em);
668 
669  if (!q_fail->pert_cons) {
670 
671  //
672  // the reduction failed because there was more than one block
673  // of pertinent children. The reduction in this case assures that
674  // pert_begin and pert_end lie in different blocks and that
675  // --pert_end is empty and lies between these two blocks.
676  //
677  // This is one of the two cases that may apply when the reduction
678  // fails already in bubble up. The reduction takes care of this.
679  //
680 
681  GTL_debug::debug_message ("CASE C (non consecutive pertinent children)\n");
682  pq_tree::sons_iterator tmp = q_fail->pert_begin;
683  pq_leaf* leaves[3];
684  node nodes[3];
685  leaves[0] = search_full_leaf (*tmp);
686  nodes[0] = (*tmp)->up;
687 
688  tmp = q_fail->pert_end;
689  leaves[2] = search_full_leaf (*tmp);
690  nodes[2] = (*tmp)->up;
691 
692  --tmp;
693  while ((*tmp)->kind() == pq_node::DIR) {
694  --tmp;
695  }
696 
697  leaves[1] = search_empty_leaf (*tmp);
698  nodes[1] = (*tmp)->up;
699 
700  case_C (nodes, leaves, st_, to_father, G, q_fail);
701 
702  } else if (!(*(q_fail->pert_end))->is_endmost && !is_root) {
703 
704  GTL_debug::debug_message ("CASE D (non-root q-node with both endmost sons empty)\n");
705  pq_tree::sons_iterator tmp = q_fail->sons.begin();
706  pq_leaf* leaves[3];
707  node nodes[3];
708  leaves[0] = search_empty_leaf (*tmp);
709  nodes[0] = (*tmp)->up;
710 
711  tmp = --(q_fail->sons.end());
712  leaves[2] = search_empty_leaf (*tmp);
713  nodes[2] = (*tmp)->up;
714 
715  tmp = q_fail->pert_begin;
716  leaves[1] = search_full_leaf (*tmp);
717  nodes[1] = (*tmp)->up;
718 
719  case_D (nodes, leaves, st_, to_father, G, q_fail);
720 
721  } else if (q_fail->partial_count == 1) {
722  if (q_fail->partial_pos[0] == q_fail->pert_end) {
723  GTL_debug::debug_message ("CASE D (non-root q-node with partial child at end of pertinent children\n");
724  pq_tree::sons_iterator tmp = q_fail->sons.begin();
725  pq_leaf* leaves[3];
726  node nodes[3];
727  leaves[0] = search_empty_leaf (*tmp);
728  nodes[0] = (*tmp)->up;
729 
730  tmp = q_fail->pert_end;
731  leaves[2] = search_empty_leaf (*tmp);
732  nodes[2] = (*tmp)->up;
733
