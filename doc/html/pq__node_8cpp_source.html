122 void q_node::pertinent (iterator it)
123 {
124  iterator end = sons.end();
125  iterator tmp = it;
126  pq_node* first;
127  pq_node* last;
128  pert_end = it;
129  ++tmp;
130  int pert_block_count = 1;
131 
132  while (tmp != end) {
133  if ((*tmp)->mark != UNBLOCKED) {
134  break;
135  }
136 
137  if ((*tmp)->kind () != DIR) {
138  ++pert_block_count;
139  pert_end = tmp;
140  }
141 
142  ++tmp;
143  }
144 
145  last = *pert_end;
146 
147  pert_begin = tmp = it;
148  --tmp;
149 
150  while (tmp != end) {
151  if ((*tmp)->mark != UNBLOCKED) {
152  break;
153  }
154 
155  if ((*tmp)->kind () != DIR) {
156  ++pert_block_count;
157  pert_begin = tmp;
158  }
159 
160  --tmp;
161  }
162 
163  first = *pert_begin;
164  pert_cons = (pert_block_count == pert_children);
165 
166  //
167  // it must be true, that either first or last is in
168  // {sons.front(), sons.back()} (or both). Thus we are able to achive
169  // the following normalization: pert_end is *always* sons.last() and
170  // pert_begin is some other child, such that ++pert_begin leads towards
171  // pert_end.
172  //
173 
174  if (pert_cons) {
175  if (last == sons.front()) {
176  turn();
177  } else if (last != sons.back()) {
178  tmp = pert_begin;
179  pert_begin = pert_end;
180  pert_end = tmp;
181  pert_end.reverse();
182  pert_begin.reverse();
183 
184  if (first == sons.front()) {
185  turn();
186  } else if (first != sons.back()) {
187 
188  //
189  // This should not happen. In this case the pertinent children are
190  // BLOCKED and thus this method would´t be called.
191  //
192  // 17.3. Now this can happen.
193  //
194 
195  // pert_cons = false;
196 
197  // assert (false);
198  }
199  }
200 
201  } else {
202 
203  //
204  // In case that there are more than one block of pertinent children although
205  // bubble up didn´t fail (e.g. pp...pe...ep...pp or ee...ep...pe...ep...pp)
206  // we need some element of the second block in order to find K5 or K33. So we
207  // leave pert_begin as it is, but assign pert_end to something in the second
208  // block
209  //
210 
211  tmp = pert_begin;
212  --tmp;
213 
214  while (tmp != sons.end()) {
215  if ((*tmp)->mark == UNBLOCKED && (*tmp)->kind () != DIR) {
216  break;
217  }
218 
219  --tmp;
220  }
221 
222 
223  //
224  // We need an empty child. So we always keep the invariant that --pert_end
225  // leads to an empty child. Please note that --pert_end might be a DI.
226  //
227 
228  tmp.reverse();
229 
230  if (tmp == sons.end()) {
231  tmp = pert_end;
232  ++tmp;
233 
234  while (tmp != sons.end()) {
235  if ((*tmp)->mark == UNBLOCKED && (*tmp)->kind () != DIR) {
236  break;
237  }
238 
239  ++tmp;
240  }
241 
242  assert (tmp != sons.end());
243  }
244 
245  pert_end = tmp;
246  }
247 
248  //
249  // In the case that there is in fact only one pertinent child we so far
250  // only assured that it is the last child, but it is still possible
251  // that pert_begin (and pert_end, too) is headed the wrong way.
252  //
253 
254  if (pert_begin == pert_end && pert_cons && pert_end == --(sons.end())) {
255  pert_begin = pert_end = --(sons.end());
256  }
257 }

282 q_node* q_node::merge (iterator it)
283 {
284  assert ((*it)->kind() == pq_node::Q_NODE);
285  q_node* part = (q_node*) *it;
286 
287  if (part == sons.front()) {
288  part->sons.front()->father = this;
289  part->sons.back()->is_endmost = false;
290  } else if (part == sons.back()){
291  part->sons.back()->father = this;
292  part->sons.front()->is_endmost = false;
293  } else {
294  part->sons.front()->is_endmost = false;
295  part->sons.back()->is_endmost = false;
296  }
297 
298  sons.splice (it, part->sons.begin(), part->sons.end());
299  sons.erase (it);
300 
301  return part;
302 }
