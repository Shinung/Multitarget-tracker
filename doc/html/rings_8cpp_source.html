136  // We look at the sorted list of children and
137  // put the split between inner and outer rings at the point where
138  // the fraction of children yet to be included is less than the
139  // amount of space available in an inner ring if count rings are

154

156

161  inner = outer - count;
162  outer = count;
163  }
164  }

183  }
