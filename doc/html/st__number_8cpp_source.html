60 void pathfinder::dfs_sub (node& curr, node& father)
61 {
62  low_num[curr] = dfs_num[curr] = act_dfs_num++;
63  new_nodes--;
64 
65  node::adj_edges_iterator it = curr.adj_edges_begin();
66  node::adj_edges_iterator end = curr.adj_edges_end();
67 
68  while (it != end) {
69  edge adj = *it;
70  node opp = curr.opposite(adj);
71 
72  if (dfs_num[opp] == 0) {
73 
74  edges_t::iterator tmp = tree[curr].insert (tree[curr].end(), adj);
75  to_father[opp] = tmp;
76 
77  dfs_sub (opp, curr);
78 
79  if (low_num[opp] < low_num[curr]) {
80  low_num[curr] = low_num[opp];
81  to_low[curr] = tmp;
82  }
83 
84  if (low_num[opp] >= dfs_num[curr]) {
85  is_biconn = false;
86  }
87 
88  } else if (opp != father && dfs_num[opp] < dfs_num[curr]) {
89  edges_t::iterator back_pos = back[curr].insert (back[curr].end(), adj);
90  edges_t::iterator forward_pos = forward[opp].insert (forward[opp].end(), adj);
91  pos[adj] = pos_pair (forward_pos, back_pos);
92 
93  if (dfs_num[opp] < low_num[curr]) {
94  low_num[curr] = dfs_num[opp];
95  to_low[curr] = back_pos;
96  }
97  }
98 
99  ++it;
100  }
101 }

212 {

219

223  nodes_t::iterator pos;
224 
225  if (it == end)
226  {
227  st_num[tmp] = act_st++;
228  st_ord.push_back(tmp);
229  tmp = order.back();
230  order.pop_back();
231 
232  }
233  else

235  pos = order.end();
236 
237  while (it != end)
238  {
239  pos = order.insert(pos, *it);

241  }
242

244  }
245  }
246
